Contoh jQuery

Pengenalan

jQuery adalah pustaka JavaScript yang cepat, kecil, dan kaya fitur. jQuery menyederhanakan berbagai hal seperti traversal dan manipulasi dokumen HTML, penanganan event, dan animasi. jQuery mudah digunakan dan membuat bekerja dengan JavaScript menjadi lebih intuitif.

Contoh Kode

Berikut adalah contoh skrip jQuery sederhana:


    $(document).ready(function () {
        var countdownTimer;
                
        $("#startButton").click(function () {
            var inputDate = $("#datetime").val();
            if (!inputDate) {
                alert("Silakan pilih waktu akhir.");
                return;
            }
                
            var countDownDate = new Date(inputDate).getTime();
                
            if (countdownTimer) {
                            clearInterval(countdownTimer);
            }
                
             countdownTimer = setInterval(function () {
                var now = new Date().getTime();
                var distance = countDownDate - now;
                
                var hari = Math.floor(distance / (1000 * 60 * 60 * 24));
                var jam = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                var menit = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
                var detik = Math.floor((distance % (1000 * 60)) / 1000);
                
                $("#hari").text(hari);
                $("#jam").text(jam);
                $("#menit").text(menit);
                $("#detik").text(detik);
                $("#countdown").show();
                
                if (distance < 0) {
                    clearInterval(countdownTimer);
                    $("#countdown").text("Waktu Habis");
                }
            }, 1000);
        });
    });
